167 inline
168 double
169 error_single_param(const double &param, const cv_strategy_t &strat, const std::size_t &number_cv_iter)
170 const
171 {
172 double err = 0.0;
173
174 //if OMP: going parallel
175#ifdef _OPENMP
176#pragma omp parallel for shared(param,strat) num_threads(this->number_threads()) reduction(+:err)
177 for (int i = 0; i < number_cv_iter; ++i)
178 {
179 auto train_valid_set = this->strategy().train_validation_set(this->Data(),strat[i]);
180 err += this->error_single_cv_iter(param,train_valid_set.first,train_valid_set.second);
181 }
182#else
183 // if not OMP: STL algorithms: transform_reduce to apply at each split the evaluation of the prediction and then summing them up
184 err = std::transform_reduce(strat.cbegin(),
185 strat.cend(),
186 0.0,
187 std::plus{},
188 [this,&param](auto el){ auto train_valid_set = this->strategy().train_validation_set(this->Data(),el); return this->error_single_cv_iter(param,train_valid_set.first,train_valid_set.second);});
189#endif
190
191 //returning the average
192 return err/(static_cast<double>(number_cv_iter));
193 }

200 inline
201 void
202 best_param_search()
203 {
204 int tot_params = m_params.size();
205
206 //preparing the container for the errors: resize to use transform
207 m_valid_errors.resize(tot_params);
208
209 //if OMP: going parallel
210#ifdef _OPENMP
211#pragma omp parallel for shared(m_params,tot_params) num_threads(this->number_threads())
212 for(std::size_t i = 0; i < tot_params; ++i)
213 {
214 m_valid_errors[i] = this->error_single_param(m_params[i],this->strategy().strategy(),this->strategy().strategy().size());
215 }
216#else
217 // if not OMP: STL algorithms: transform to apply at each element the input space the function to retain their validation error
218 std::transform(m_params.cbegin(),
219 m_params.cend(),
220 m_valid_errors.begin(),
221 [this](double const &param_i){return this->error_single_param(param_i,this->strategy().strategy(),this->strategy().strategy().size());});
222#endif
223
224 //best validation error
225 auto min_err = (std::min_element(m_valid_errors.begin(),m_valid_errors.end()));
226 m_best_valid_error = *min_err;
227
228 //optimal param
229 m_param_best = m_params[std::distance(m_valid_errors.begin(),min_err)];
230
231 //free memory
232 if constexpr(valid_err_ret == VALID_ERR_RET::NO_err)
233 {
234 m_valid_errors.clear();
235 }
236 }
