37template< SOLVER solver, K_IMP k_imp, VALID_ERR_RET valid_err_ret, CV_STRAT cv_strat, CV_ERR_EVAL cv_err_eval >
38void
39PPC_KO_wrapper_no_cv<solver,k_imp,valid_err_ret,cv_strat,cv_err_eval>::call_ko()
40{
41
42 if constexpr(k_imp == K_IMP::YES) //k imposed
43 {
44 //class for computations construction
45 PPC_KO_NoCV<solver,K_IMP::YES,valid_err_ret,cv_strat,cv_err_eval> KO(std::move(this->data()),m_alpha,m_k,this->number_threads());
46 //solving
47 KO.solve();
48 //computing scores
49 auto scores = KO.scores();
50 //computing sd of scores of directions and weights
51 auto sd_scores = KO.sd_scores_dir_wei();
52
53 //if validation errors have to be stored and returned
54 if constexpr( valid_err_ret == VALID_ERR_RET::YES_err){this->results() = std::make_tuple(KO.prediction(),KO.alpha(),KO.k(),scores,KO.explanatory_power(),KO.a(),KO.b(),sd_scores,KO.means(),KO.ValidErr());}
55 else {this->results() = std::make_tuple(KO.prediction(),KO.alpha(),KO.k(),scores,KO.explanatory_power(),KO.a(),KO.b(),sd_scores,KO.means());}
56 }
57
58 if constexpr(k_imp == K_IMP::NO) //k to be found with explanatory power criterion
59 {
60 //class for computations construction
61 PPC_KO_NoCV<solver,K_IMP::NO,valid_err_ret,cv_strat,cv_err_eval> KO(std::move(this->data()),m_alpha,m_threshold_ppc,this->number_threads());
62 //solving
63 KO.solve();
64 //computing scores
65 auto scores = KO.scores();
66 //computing sd of scores of directions and weights
67 auto sd_scores = KO.sd_scores_dir_wei();
68
69 //if validation errors have to be stored and returned
70 if constexpr( valid_err_ret == VALID_ERR_RET::YES_err){this->results() = std::make_tuple(KO.prediction(),KO.alpha(),KO.k(),scores,KO.explanatory_power(),KO.a(),KO.b(),sd_scores,KO.means(),KO.ValidErr());}
71 else {this->results() = std::make_tuple(KO.prediction(),KO.alpha(),KO.k(),scores,KO.explanatory_power(),KO.a(),KO.b(),sd_scores,KO.means());}
72 }
73}
