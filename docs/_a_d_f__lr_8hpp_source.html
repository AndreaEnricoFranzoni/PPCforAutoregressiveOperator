77 inline void std_er_coeff(const arma::mat &x)
78 {
79 //covariates rank
80 auto r = rank(x);
81
82 //full rank: x*x' symmetric and invertible
83 if(r == std::min(x.n_rows,x.n_cols))
84 {
85 //using Armadillo objects: creating x*x'
86 arma::mat inv = arma::inv_sympd(x*trans(x));
87 //inverse
88 arma::vec diag_inv = arma::diagvec(inv);
89 //mapping back to Eigen the standard errors
90 m_se_coeff = Eigen::Map<KO_Traits::StoringVector>(diag_inv.memptr(), diag_inv.n_elem);
91 m_se_coeff = m_rse*m_se_coeff;
92 //square root
93 std::for_each(m_se_coeff.begin(), m_se_coeff.end(), [](auto &el){el=std::sqrt(el);});
94 }
95 //not full rank: use pseudo inverse
96 else
97 {
98 //using Armadillo objects: creating x*x'
99 arma::mat inv = arma::pinv(x*trans(x));
100 //pseudo-inverse
101 arma::vec diag_inv = arma::diagvec(inv);
102 //mapping back to Eigen the standard errors
103 m_se_coeff = Eigen::Map<KO_Traits::StoringVector>(diag_inv.memptr(), diag_inv.n_elem);
104 m_se_coeff = m_rse*m_se_coeff;
105 //square root
106 std::for_each(m_se_coeff.data(), m_se_coeff.data() + m_se_coeff.size(), [](auto &el){el=std::sqrt(el);});
107 }
108 }

114 inline void solve()
115 {
116 //convert covariates and responses (Eigen objects) to Armadillo ones in order to use mlpack
117 arma::mat covariates(m_x.data(), m_x.rows(), m_x.cols(), false, true);
118 arma::rowvec responses(m_y.data(), m_y.rows(), false, true);
119
120 //train the model of linear regression (including intercept)
121 mlpack::LinearRegression lr(covariates,responses);
122
123 //for adf: needed estaimate of the coefficients and standard errors of these estimates
124 //estimate of the coefficients
125 m_coeff = Eigen::Map<KO_Traits::StoringVector>(lr.Parameters().memptr(),
126 lr.Parameters().n_rows,
127 lr.Parameters().n_cols);
128
129 //mse between responses and fitted values
130 m_mse_res = lr.ComputeError(covariates, responses);
131
132 //rse: mse on residulas divided by df (statistical units - covariates (excluding intercept))
133 //estimate of sigma_2
134 m_rse = m_mse_res*(static_cast<double>(covariates.n_cols)/static_cast<double>(covariates.n_cols - (covariates.n_rows+static_cast<std::size_t>(1))));
135
136 //in order to have the standard error of the estimates of the coefficients, it is necessary to include the constant in the design matrix
137 //to be done as this in armadillo
138 arma::mat covariates_plus_intercept(covariates.n_rows + 1, covariates.n_cols);
139 covariates_plus_intercept.row(0).ones();
140 covariates_plus_intercept.rows(1, covariates.n_rows) = covariates;
141 //standard errors of coefficients estimates
142 this->std_er_coeff(covariates_plus_intercept);
143 }
