125 template<typename STOR_OBJ>
126 PPC_KO_base(STOR_OBJ&& X,int number_threads)
127 :
128 m_X{std::forward<STOR_OBJ>(X)},
129 m_m(X.rows()),
130 m_n(X.cols()),
131 m_number_threads(number_threads)
132 {
133 //evaluating row mean and saving it in the m_means
134 m_means = (m_X.rowwise().sum())/m_n;
135
136 //centering
137#ifdef _OPENMP
138#pragma omp parallel for num_threads(m_number_threads)
139#endif
140 for (size_t i = 0; i < m_n; ++i)
141 {
142 m_X.col(i) = m_X.col(i).array() - m_means;
143 }
144
145 // covariance operator estimate: (X * X')/n
146 m_Cov = ((m_X*m_X.transpose()).array())/static_cast<double>(m_n);
147
148 // trace of covariance
149 m_trace_cov = m_Cov.trace();
150
151 // cross-covariance operator estimate: (X[,2:n]*(X[,1:(n-1)])')/(n-1)
152 m_CrossCov = ((m_X.rightCols(m_n-1)*m_X.leftCols(m_n-1).transpose()).array())/(static_cast<double>(m_n-1));
153
154 // square of cross covariance estimate
155 m_GammaSquared = m_CrossCov.transpose()*m_CrossCov;
156 }
