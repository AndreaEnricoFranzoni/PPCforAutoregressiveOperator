141 inline
142 double
143 error_single_param(const int &param, const cv_strategy_t &strat, const std::size_t &number_cv_iter)
144 const
145 {
146 double err = 0.0;
147
148 //if OMP: going parallel
149#ifdef _OPENMP
150#pragma omp parallel for shared(param,strat) num_threads(this->number_threads()) reduction(+:err)
151 for (int i = 0; i < number_cv_iter; ++i)
152 {
153 auto train_valid_set = this->strategy().train_validation_set(this->Data(),strat[i]);
154 err += this->error_single_cv_iter(param,train_valid_set.first,train_valid_set.second);
155 }
156#else
157 // if not OMP: STL algorithms: transform_reduce to apply at each split the evaluation of the prediction and then summing them up
158 err = std::transform_reduce(strat.cbegin(),
159 strat.cend(),
160 0.0,
161 std::plus{},
162 [this,&param](auto el){ auto train_valid_set = this->strategy().train_validation_set(this->Data(),el); return this->error_single_cv_iter(param,train_valid_set.first,train_valid_set.second);});
163#endif
164
165 //returning the average
166 return err/(static_cast<double>(number_cv_iter));
167 }

173 inline
174 void
175 best_param_search()
176 {
177 //preparing the container for the validation errors
178 m_valid_errors.reserve(m_params.size());
179
180 //evaluating validation error for each parameter
181 double previous_error(static_cast<double>(0));
182
183 //if adding another PPC does not improve too much the validation error: break
184 for(const auto & el : m_params)
185 {
186 //evaluate the error for the parameter
187 double curr_err = this->error_single_param(el,this->strategy().strategy(),this->strategy().strategy().size());
188 m_valid_errors.emplace_back(curr_err);
189
190 if(std::abs(curr_err - previous_error) < m_toll) {break;} else {previous_error = curr_err;}
191 }
192
193 //Shrinking
194 m_valid_errors.shrink_to_fit();
195
196 //best validation error
197 auto min_err = (std::min_element(m_valid_errors.begin(),m_valid_errors.end()));
198 m_best_valid_error = *min_err;
199
200 //optimal param
201 m_param_best = m_params[std::distance(m_valid_errors.begin(),min_err)];
202
203 //if saving errors
204 if constexpr (valid_err_ret == VALID_ERR_RET::NO_err)
205 {
206 m_valid_errors.clear();
207 }
208 }
