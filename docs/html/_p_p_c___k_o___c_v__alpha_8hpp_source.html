131 solving()
132 {
133 //factory to create the cv strategy
134 auto strategy_cv = Factory_cv_strat<cv_strat>::cv_strat_obj(m_min_size_ts,m_max_size_ts);
135
136 // if k imposed
137 if constexpr(k_imp == K_IMP::YES)
138 {
139 //lambda wrapper for the correct overload for prediction function
140 auto predictor = [](KO_Traits::StoringMatrix&& data, double alpha, int k, int number_threads) { return cv_pred_func<solver,K_IMP::YES,VALID_ERR_RET::NO_err,cv_strat,cv_err_eval>(std::move(data),alpha,k,number_threads);};
141
142 //cv knowing k
143 CV_alpha<cv_strat,cv_err_eval,k_imp,valid_err_ret> cv(std::move(m_X_non_cent),std::move(*strategy_cv),m_alphas,this->k(),predictor,this->number_threads());
144
145 //best alpha
146 cv.best_param_search();
147 this->alpha() = cv.param_best(); //finding the best alpha using CV
148
149 //if errors to be saved
150 if constexpr(valid_err_ret == VALID_ERR_RET::YES_err)
151 {
152 valid_err_cv_1_t m;
153 m.reserve(m_alphas.size());
154 for (size_t i = 0; i < m_alphas.size(); ++i) { m.emplace_back(cv.valid_errors()[i]);}
155 this->ValidErr() = m;
156 m.clear();
157 }
158 }
159
160 // if k selected through explanatory power criterion
161 if constexpr(k_imp == K_IMP::NO)
162 {
163 //lambda wrapper for the correct overload for prediction function
164 auto predictor = [](KO_Traits::StoringMatrix&& data, double alpha, double threshold_ppc, int number_threads) { return cv_pred_func<solver,K_IMP::NO,VALID_ERR_RET::NO_err,cv_strat,cv_err_eval>(std::move(data),alpha,threshold_ppc,number_threads);};
165
166 //cv with k to be found with explanatory power
167 CV_alpha<cv_strat,cv_err_eval,k_imp,valid_err_ret> cv(std::move(m_X_non_cent),std::move(*strategy_cv),m_alphas,this->threshold_ppc(),predictor,this->number_threads());
168
169 //best alpha
170 cv.best_param_search();
171 this->alpha() = cv.param_best(); //finding the best alpha using CV
172
173 //only if errors are saved
174 if constexpr(valid_err_ret == VALID_ERR_RET::YES_err)
175 {
176 valid_err_cv_1_t m;
177 m.reserve(m_alphas.size());
178 for (size_t i = 0; i < m_alphas.size(); ++i) { m.emplace_back(cv.valid_errors()[i]);}
179 this->ValidErr() = m;
180 m.clear();
181 }
182 }
183
184 //computing regularized covariance
185 this->CovReg() = this->Cov().array() + this->alpha()*this->trace_cov()*(KO_Traits::StoringMatrix::Identity(this->m(),this->m()).array());
186 //PPCKO
187 this->KO_algo();
188 };
