140 inline
141 void
142 best_param_search()
143 {
144 std::size_t tot_alphas = m_alphas.size();
145
146 //if OMP: going parallel
147#ifdef _OPENMP
148 //preparing the containers for the errors
149 if constexpr(valid_err_ret == VALID_ERR_RET::YES_err)
150 {
151 m_valid_errors.resize(tot_alphas); //for each alpha: valid error for each k
152 }
153
154 m_valid_errors_best_pairs.resize(tot_alphas); //for each alpha: valid error only for the best k
155
156#pragma omp parallel for shared(m_alphas,m_k_s,m_toll,m_pred_f,tot_alphas) num_threads(this->number_threads()) schedule(guided)
157 for(std::size_t i = 0; i < tot_alphas; ++i)
158 {
159 //alpha fixed: doing CV on k
160 CV_k<cv_strat,err_eval,k_imp,valid_err_ret> cv(std::move(this->Data()),std::move(this->strategy()),m_k_s,m_toll,m_alphas[i],m_pred_f,this->number_threads());
161 cv.best_param_search();
162
163 //best k given the alpha
164 m_best_pairs.insert(std::make_pair(m_alphas[i],cv.param_best()));
165 //saving the validation error for the best pair
166 m_valid_errors_best_pairs[i]=cv.best_valid_error();
167
168 if constexpr(valid_err_ret == VALID_ERR_RET::YES_err)
169 {
170 //saving the validation error for each pair (further inspection)
171 m_valid_errors[i]=cv.valid_errors();
172 }
173 }
174#else
175 //preparing the containers for the errors
176 if constexpr(valid_err_ret == VALID_ERR_RET::YES_err)
177 {
178 m_valid_errors.reserve(tot_alphas); //for each alpha: valid error for each k
179 }
180
181 m_valid_errors_best_pairs.reserve(tot_alphas); //for each alpha: valid error only for the best k
182
183 for(const auto & alpha : m_alphas)
184 {
185 //alpha fixed: doing CV on k
186 CV_k<cv_strat,err_eval,k_imp,valid_err_ret> cv(std::move(this->Data()),std::move(this->strategy()),m_k_s,m_toll,alpha,m_pred_f,this->number_threads());
187 cv.best_param_search();
188
189 //best k given the alpha
190 m_best_pairs.insert(std::make_pair(alpha,cv.param_best()));
191 //saving the validation error for the best pair
192 m_valid_errors_best_pairs.emplace_back(cv.best_valid_error());
193
194 if constexpr(valid_err_ret == VALID_ERR_RET::YES_err)
195 {
196 //saving the validation error for each pair (further inspection)
197 m_valid_errors.emplace_back(cv.valid_errors());
198 }
199 }
200#endif
201
202 //best validation error
203 auto min_err = std::min_element(m_valid_errors_best_pairs.begin(),m_valid_errors_best_pairs.end());
204 m_best_valid_error = *min_err;
205
206 //best alpha
207 m_alpha_best = m_alphas[std::distance(m_valid_errors_best_pairs.begin(),min_err)];
208
209 //best k
210 m_k_best = m_best_pairs.find(m_alpha_best)->second;
211 }
