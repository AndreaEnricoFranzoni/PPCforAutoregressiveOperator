37template<class LAG_policy>
38std::vector<double>
39adf<LAG_policy>::one_step_diff(const KO_Traits::StoringVector &ts)
40const
41{
42 //compute the differences within adjacent values of the time series
43 std::vector<double> diff;
44 diff.resize(m_tot_time_instants-1);
45 std::adjacent_difference(ts.cbegin()+1,ts.cend(),diff.begin());
46 diff[0] = ts[1] - ts[0];
47
48 return diff;
49}

61template<class LAG_policy>
62KO_Traits::StoringMatrix
63adf<LAG_policy>::embed(const KO_Traits::StoringVector &ts, int dimension)
64const
65{
66 //evaluate one step differences for the time serie
67 std::vector<double> y(this->one_step_diff(ts));
68
69 int n = y.size();
70 int m = n - dimension + 1; //number of rows of embedded ts
71
72 KO_Traits::StoringMatrix embedded(m,dimension);
73
74 //embedding
75 std::vector<std::size_t> start_ind;
76 start_ind.resize(static_cast<std::size_t>(m));
77 std::iota(start_ind.begin(),start_ind.end(),static_cast<std::size_t>(0));
78
79 for(std::size_t i = 0; i < static_cast<std::size_t>(dimension); ++i)
80 {
81 std::size_t gain = static_cast<std::size_t>(dimension) - i - 1;
82
83 std::transform(start_ind.begin(),start_ind.end(),
84 embedded.col(i).begin(),
85 [&y,&gain](std::size_t el){ return y[el+gain];});
86
87 }
88
89 start_ind.clear();
90
91 return embedded;
92}

101template<class LAG_policy>
102double
103adf<LAG_policy>::statistic_eval(const KO_Traits::StoringVector &ts)
104const
105{
106 //embedding of the time serie in a low-dimensional Euclidean space
107 KO_Traits::StoringMatrix z = this->embed(ts,m_k_used);
108
109 //the staistics is computed evaluating coefficients standard errors of a fitted linear regression
110 //the design matrix depends on the time lag: Lag_policy policy accounts for it
111 adf_stat<LAG_policy> statistic_adf;
112
113 //evaluate the statistic
114 return statistic_adf(ts,z);

126template<class LAG_policy>
127double
128adf<LAG_policy>::p_value_eval(const KO_Traits::StoringVector &ts, const std::vector<double> &tableipl, int i)
129const
130{
131 //evaluation of the test statistic
132 double stat = this->statistic_eval(ts);
133
134 //if statistic too extreme the pvalue is put to 0 or 1
135 if(stat<tableipl.front()){return 0.0;}
136 if(stat>tableipl.back()){return 1.0;}
137
138 //pvalue evaluation
139 interp_func p_val_int{tableipl}; //retain the p-value using interpolation: preparing the functor, with the discrete evaluations available
140 double pval = p_val_int(stat,tablep); //performing interpolation
141
142 return pval;
143}
